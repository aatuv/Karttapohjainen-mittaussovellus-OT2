CREATE TABLE anturi (
  ID int NOT NULL IDENTITY(1,1) PRIMARY KEY,
  NAME varchar(60) NOT NULL UNIQUE,
  DESCRIPTION varchar(60) NOT NULL,
);

CREATE TABLE kartta (
  ID int NOT NULL IDENTITY(1,1) PRIMARY KEY,
  NAME varchar(60) NOT NULL UNIQUE,
  PATH varchar(2083) NOT NULL,
);

CREATE TABLE mittaus (
  ID int NOT NULL IDENTITY(1,1) PRIMARY KEY,
  ANTURI_ID int REFERENCES anturi (ID),
  TIMESTAMP datetime DEFAULT CURRENT_TIMESTAMP,
  NOTE varchar(60) NOT NULL,
);
CREATE TABLE data (
  ID int NOT NULL IDENTITY(1,1) PRIMARY KEY,
  MITTAUS_ID int REFERENCES mittaus (ID),
  TEMPERATURE decimal(3,1) DEFAULT NULL,
  ATMOSP_PRESSURE decimal(6,2) DEFAULT NULL,
  REL_AIR_HUMIDITY decimal(4,4) DEFAULT NULL,
);
CREATE TABLE anturi_sijainti (
  KARTTA_ID int REFERENCES kartta (ID),
  ANTURI_ID int REFERENCES anturi (ID),
  X float DEFAULT 0 NOT NULL,
  Y float DEFAULT 0 NOT NULL, 
);
